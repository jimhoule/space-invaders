cmake_minimum_required(VERSION 3.29.5)

project(SpaceInvaders VERSION 1.0.0 LANGUAGES C CXX)

set(CMAKE_CXX_STANDARD 14)
set(CMAKE_Cxx_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# Creates a variable
set(SPACE_INVADERS_GAME_TARGET_NAME SpaceInvadersGame)
set(SFML_LIB_NAME Sfml)

# Allows fetching content from remote location
include(FetchContent)
# Fetches contents
FetchContent_Declare(
    ${SFML_LIB_NAME}
    GIT_REPOSITORY https://github.com/SFML/SFML.git
    GIT_TAG 2.6.1 GIT_SHALLOW FALSE
)
FetchContent_MakeAvailable(${SFML_LIB_NAME})

# Adds subdirectories
add_subdirectory(SpaceInvaders)
